use msdb
go 

select m.name as 'job_名稱'
,m.enabled
,d.command as'步驟語法'
,d.step_id as '步驟順序'
,d.step_name as '步驟名稱'
,m.description as '描述'
from dbo.sysjobs as m 
left join dbo.sysjobsteps as d 
on m.job_id = d.job_id
where enabled = 1
--and (d.command like 'exec%' or d.command like 'EXEC%' or d.command like '%exec%' or d.command like '%EXEC%')
order by m.name
